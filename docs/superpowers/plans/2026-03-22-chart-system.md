# Chart System Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `@rokkit/chart` — a data-driven SVG chart library with ggplot-style builder API, ChartBrewer context, palette-based colors, and full theme CSS support.

**Architecture:** Three-layer system: `ChartSpec` mutable builder → `ChartBrewer` Svelte 5 reactive class (colors, patterns, D3 scales, mark geometry) → `Chart.svelte` wrapper + individual chart components + `Plot.*` primitives. Charts use `data-chart-*` attributes for theme CSS and inline SVG fill/stroke from a 21-color `palette.json`.

**Tech Stack:** Svelte 5 (runes), d3-scale, d3-array, d3-axis, UnoCSS (stroke/fill tone utilities), Vitest, Playwright

```js
import palette from './palette.json'

/**
 * Extracts distinct values for a given field from the data array.
 * @param {Object[]} data
 * @param {string} field
 * @returns {unknown[]}
 */
export function distinct(data, field) {
  if (!field) return []
  return [...new Set(data.map((d) => d[field]))]
}

/**
 * Assigns palette colors to an array of distinct values.
 * @param {unknown[]} values
 * @param {'light'|'dark'} mode
 * @returns {Map<unknown, {fill: string, stroke: string}>}
 */
export function assignColors(values, mode = 'light') {
  return new Map(values.map((v, i) => [v, palette[i % palette.length].shades[mode]]))
}
```

```js
export class ChartSpec {
  constructor(data) {
    this.data = data ?? []
    this.channels = {}
    this.layers = []
    this.options = {}
  }

  x(f) {
    this.channels.x = f
    return this
  }
  y(f) {
    this.channels.y = f
    return this
  }
  color(f) {
    this.channels.color = f
    return this
  }
  pattern(f) {
    this.channels.pattern = f
    return this
  }
  aes(ch) {
    Object.assign(this.channels, ch)
    return this
  }

  bar(opts = {}) {
    this.layers.push({ type: 'bar', ...opts })
    return this
  }
  line(opts = {}) {
    this.layers.push({ type: 'line', ...opts })
    return this
  }
  area(opts = {}) {
    this.layers.push({ type: 'area', ...opts })
    return this
  }
  arc(opts = {}) {
    this.layers.push({ type: 'arc', ...opts })
```

```js
import { scaleBand, scaleLinear, scaleSqrt } from 'd3-scale'
import { max, extent } from 'd3-array'

/**
 * Builds an x scale (band for categorical, linear for numeric).
 * @param {Object[]} data
 * @param {string} field
 * @param {number} width - inner width (pixels)
 * @param {{ padding?: number }} opts
 */
export function buildXScale(data, field, width, opts = {}) {
  const values = [...new Set(data.map((d) => d[field]))]
  const isNumeric = values.every((v) => typeof v === 'number' || !isNaN(Number(v)))
  if (isNumeric) {
    const [min, maxVal] = extent(data, (d) => d[field])
    return scaleLinear().domain([min, maxVal]).range([0, width]).nice()
  }
  return scaleBand()
    .domain(values)
    .range([0, width])
    .padding(opts.padding ?? 0.2)
}

/**
 * Builds a y linear scale from 0 to max, extended by any layer overrides.
 * @param {Object[]} data
 * @param {string} field
 * @param {number} height - inner height (pixels)
 * @param {Array<{data?: Object[], y?: string}>} layers - additional layers that may extend domain
 */
export function buildYScale(data, field, height, layers = []) {
  let maxVal = max(data, (d) => +d[field]) ?? 0
  for (const layer of layers) {
    if (layer.data && layer.y) {
      const layerMax = max(layer.data, (d) => +d[layer.y]) ?? 0
      if (layerMax > maxVal) maxVal = layerMax
    }
  }
  return scaleLinear().domain([0, maxVal]).range([height, 0]).nice()
}

/**
 * Builds a sqrt scale for bubble/point size.
 * @param {Object[]} data
 * @param {string} field
 * @param {number} maxRadius
 */
export function buildSizeScale(data, field, maxRadius = 20) {
  const maxVal = max(data, (d) => +d[field]) ?? 1
  return scaleSqrt().domain([0, maxVal]).range([0, maxRadius])
}
```

```svelte
  // @ts-nocheck
  import { scaleLinear } from 'd3-scale'
  import { max, min } from 'd3-array'

  /** @type {number[] | object[]} */
  let {
    data = [],
    field = null,
    type = 'line',
    color = 'primary',
    width = 80,
    height = 24,
    min: minProp = null,
    max: maxProp = null
  } = $props()

  const values = $derived(field ? data.map((d) => +d[field]) : data.map((n) => +n))

  const yMin = $derived(minProp ?? min(values) ?? 0)
  const yMax = $derived(maxProp ?? max(values) ?? 1)

  const xScale = $derived(
    scaleLinear()
      .domain([0, Math.max(values.length - 1, 1)])
      .range([0, width])
  )
  const yScale = $derived(scaleLinear().domain([yMin, yMax]).range([height, 0]))

  const linePath = $derived(() => {
    if (values.length < 2) return ''
    return values.map((v, i) => `${i === 0 ? 'M' : 'L'}${xScale(i)},${yScale(v)}`).join(' ')
  })

  const areaPath = $derived(() => {
    if (values.length < 2) return ''
    const pts = values.map((v, i) => `${xScale(i)},${yScale(v)}`).join(' L')
    return `M${pts} L${xScale(values.length - 1)},${height} L0,${height} Z`
  })

  const barWidth = $derived(width / Math.max(values.length, 1))
</script>

<svg
  data-sparkline
  data-color={color}
  {width}
  {height}
```

```svelte
  // @ts-nocheck
  import { setContext, onMount } from 'svelte'
  import { ChartBrewer } from './lib/brewing/brewer.svelte.js'
  import { ChartSpec } from './spec/chart-spec.js'
  import Root from './Plot/Root.svelte'

  let {
    // Declarative props (generates ChartSpec internally)
    data = null,
    x = null,
    y = null,
    color = null,
    pattern = null,
    // Programmatic spec
    spec = null,
    // Layout
    width = null,
    height = 300,
    responsive = true,
    // Appearance
    mode = 'light',
    children
  } = $props()

  const brewer = new ChartBrewer()
  setContext('chart-brewer', brewer)

  // Resolve final spec from props or spec=
  const resolvedSpec = $derived(() => {
    if (spec) return spec
    if (data) {
      const s = new ChartSpec(data)
      if (x) s.x(x)
      if (y) s.y(y)
      if (color) s.color(color)
      if (pattern) s.pattern(pattern)
      return s
    }
    return null
  })

  $effect(() => {
    const s = resolvedSpec()
    if (!s) return
    brewer.update({
      data: s.data,
      channels: s.channels,
      layers: s.layers,
      width: width ?? 600,
      height,
      mode
    })
  })

  let containerEl = $state(null)

  onMount(() => {
    if (!responsive || !containerEl) return
    const ro = new ResizeObserver(([entry]) => {
      brewer.update({ width: entry.contentRect.width })
    })
    ro.observe(containerEl)
    return () => ro.disconnect()
  })
</script>

<div

  bind:this={containerEl}
  style:width={responsive ? '100%' : `${width ?? 600}px`}
  style:height="{height}px"
```
